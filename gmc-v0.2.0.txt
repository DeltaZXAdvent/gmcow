rule("sub menuctln") {
	event {Subroutine; menuctln;}
	actions {
		Wait Until(!Is Button Held(Event Player, Button(Primary Fire)), 99999);
		While(True);
			Wait Until(Is Button Held(Event Player, Button(Primary Fire)), 99999);
			If(!Event Player.menuopen);
				Abort;
			End;
			"increment"
			While(True);
				Event Player.menuitemsel += 2;
				If(Event Player.menuitemsel >= Count Of(event player.menu[Event Player.menusel]));
					Event Player.menuitemsel = 0;
				End;
				If(event player.menu[Event Player.menusel][Event Player.menuitemsel + 1] != 0);
					Break;
				End;
			End;
			Start Rule(menuupdate, Do Nothing);
			Wait Until(!Is Button Held(Event Player, Button(Primary Fire)), 0.320);
			If(Is Button Held(Event Player, Button(Primary Fire)));
				While(True);
					"increment"
					While(True);
						Event Player.menuitemsel += 2;
						If(Event Player.menuitemsel >= Count Of(event player.menu[Event Player.menusel]));
							Event Player.menuitemsel = 0;
						End;
						If(event player.menu[Event Player.menusel][Event Player.menuitemsel + 1] != 0);
							Break;
						End;
					End;
					Start Rule(menuupdate, Do Nothing);
					Wait Until(!Is Button Held(Event Player, Button(Primary Fire)), 0.064);
					If(!Is Button Held(Event Player, Button(Primary Fire)));
						Loop;
					End;
				End;
			End;
		End;
	}
}

rule("sub menuctlp") {
	event {Subroutine; menuctlp;}
	actions {
		Wait Until(!Is Button Held(Event Player, Button(Secondary Fire)), 99999);
		While(True);
			Wait Until(Is Button Held(Event Player, Button(Secondary Fire)), 99999);
			If(!Event Player.menuopen);
				Abort;
			End;
			While(True);
				Event Player.menuitemsel -= 2;
				If(Event Player.menuitemsel < 0);
					Event Player.menuitemsel = Count Of(event player.menu[Event Player.menusel]) - 2;
				End;
				If(event player.menu[Event Player.menusel][Event Player.menuitemsel + 1] != 0);
					Break;
				End;
			End;
			Start Rule(menuupdate, Do Nothing);
			Wait Until(!Is Button Held(Event Player, Button(Secondary Fire)), 0.320);
			If(Is Button Held(Event Player, Button(Secondary Fire)));
				While(True);
					"decrement"
					While(True);
						Event Player.menuitemsel -= 2;
						If(Event Player.menuitemsel < 0);
							Event Player.menuitemsel = Count Of(event player.menu[Event Player.menusel]) - 2;
						End;
						If(event player.menu[Event Player.menusel][Event Player.menuitemsel + 1] != 0);
							Break;
						End;
					End;
					Start Rule(menuupdate, Do Nothing);
					Wait Until(!Is Button Held(Event Player, Button(Secondary Fire)), 0.064);
					If(!Is Button Held(Event Player, Button(Secondary Fire)));
						Loop;
					End;
				End;
			End;
		End;
	}
}

rule("sub menuctlsel") {
	event {Subroutine; menuctlsel;}
	actions {
		Wait Until(Is Button Held(Event Player, Button(Melee)), 99999);
		If(!Event Player.menuopen);
			Abort;
		End;
		Event Player.cmd = event player.menu[Event Player.menusel][Event Player.menuitemsel + 1];
		If(Event Player.cmd == -1);
			Event Player.cmd = event player.menu[Event Player.menusel][Event Player.menuitemsel];
		End;
		Wait Until(!Is Button Held(Event Player, Button(Melee)), 99999);
		Loop;
	}
}

rule("misc - init player") {
	event { ongoing - each player; all; all; }
	actions {
		"music channel"
		if (global.channel == 0);
			enable voice chat(event player);
			disable voice chat(event player, true, false, true);
		else if(global.channel == 1);
			enable voice chat(event player);
			disable voice chat(event player, false, true, true);
		end;

		"player title"
		Event Player.nameid[0] = custom String("{0}", Event Player);
		Event Player.nameid[1] = Global.idnxt;
		Global.idnxt += 1;
		Disable Nameplates(Event Player, All Players(All Teams));
		Disable Nameplates(All Players(All Teams), Event Player);
		Create In-World Text(Array Contains(All Heroes, Hero Of(Event Player)) && !Event Player.stat[5] ? All Players(All Teams) : Null,
			Event Player.nameid[0], Update Every Frame(Eye Position(Event Player) + Up * 0.250), 1, Clip Against Surfaces,
			Visible To Position and String, Color(White), Default Visibility);
	}
}

rule("rule - draw") {
	event {Subroutine; draw;}
	actions {
		Event Player.subflags[2] = True;
		Disallow Button(Event Player, Button(Primary Fire));
		Disallow Button(Event Player, Button(Secondary Fire));
		Start Rule(draw_, Do Nothing);
		While(True);
			While(Event Player.subflags[2] && Event Player.drawtype[2] == 1);
				"draw position default"
				Global.arg[0] = Ray Cast Hit Position(Eye Position(Event Player), Eye Position(Event Player) + Facing Direction Of(Event Player)
					* 3 * Event Player.stat[6], Null, Null, False);
				"draw position"
				Global.arg[1] = Global.arg[0];
				"max distance"
				Global.arg[2] = 0.100;
				For Global Variable(index, 0, Count Of(Global.drawentities), 1);
					If(Global.drawentities[Global.index][0] == Event Player);
						Global.arg[3] = Distance Between(Eye Position(Event Player), Global.drawentities[Global.index][2]);
						Global.arg[3] *= Sine From Degrees(Angle Between Vectors(Vector Towards(Eye Position(Event Player),
							Global.drawentities[Global.index][2]), Facing Direction Of(Event Player)));
						If(Global.arg[3] < Global.arg[2]);
							Global.arg[1] = Global.drawentities[Global.index][2];
							Global.arg[2] = Global.arg[3];
						End;
						Global.arg[3] = Distance Between(Eye Position(Event Player), Global.drawentities[Global.index][3]);
						Global.arg[3] *= Sine From Degrees(Angle Between Vectors(Vector Towards(Eye Position(Event Player),
							Global.drawentities[Global.index][3]), Facing Direction Of(Event Player)));
						If(Global.arg[3] < Global.arg[2]);
							Global.arg[1] = Global.drawentities[Global.index][3];
							Global.arg[2] = Global.arg[3];
						End;
					End;
				End;
				Event Player.drawpos = Global.arg[1];
				Wait(0.016, Ignore Condition);
			End;
			Wait Until(Event Player.drawtype[2] == 1 || !Event Player.subflags[2], 99999);
			If(!Event Player.subflags[2]);
				Break;
			End;
		End;
	}
}

rule("sub - cmddraw") {
	event {Subroutine; cmddraw;}
	actions {
		If(Global.arg[0] == 0);
			If(Event Player.subflags[2]);
				Event Player.subflags[2] = False;
				Small Message(Event Player, Custom String("关闭"));
			Else;
				Small Message(Event Player, Custom String("左键绘画右键擦除"));
				Start Rule(draw, Do Nothing);
				Start Rule(drawerase, Do Nothing);
			End;
		Else If(Global.arg[0] == 1);
			Event Player.drawtype[0] += 1;
			Event Player.drawtype[0] %= 2;
			If(Event Player.drawtype[0] == 0);
				Small Message(Event Player, Custom String("line"));
			Else If(Event Player.drawtype[0] == 1);
				Small Message(Event Player, Custom String("ball"));
			End;
		Else If(Global.arg[0] == 2);
			Event Player.drawtype[1] += 1;
			Event Player.drawtype[1] %= Count Of(Global.palette);
			Small Message(Event Player, Global.palette[Event Player.drawtype[1]]);
		Else If(Global.arg[0] == 3);
			Event Player.drawvisible = !Event Player.drawvisible;
		else if(global.arg[0] == 4);
			"fix position"
			if(event player.drawposcurrent < 0);
				global.arg[0] = -1 * event player.drawposcurrent;
				event player.drawposs[global.arg[0] - 1] =
					position of(event player);
				DISABLED small message(event player, custom string("created new position at slot {0}",
				   global.arg[0]));
				start forcing player position(event player, position of(event player), false);
				event player.drawposcurrent *= -1;
			end;
		Else If(Global.arg[0] == 5);
			"switch position"
			global.arg[0] = event player.drawposcurrent > 0 ? 1 : -1;
			event player.drawposcurrent *= global.arg[0];
			event player.drawposcurrent %= 4;
			event player.drawposcurrent += 1;
			DISABLED small message(event player, custom string("changed to position slot {0}",
				event player.drawposcurrent));
			small message(event player, event player.drawposcurrent);
			global.arg[1] = event player.drawposs[event player.drawposcurrent - 1];
			if(global.arg[0] > 0);
				if(global.arg[1] == 0);
					stop forcing player position(event player);
				else;
					start forcing player position(event player,
						global.arg[1],
						false);
				end;
			end;
			event player.drawposcurrent *= global.arg[0];
		else if(global.arg[0] == 6);
			"detach position"
			if(event player.drawposcurrent > 0);
				event player.drawposcurrent *= -1;
				stop forcing player position(event player);
			end;
		else if(global.arg[0] == 6);
			Event Player.drawtype[2] += 1;
			Event Player.drawtype[2] %= 2;
			If(Event Player.drawtype[2] == 0);
				Small Message(Event Player, Custom String("hello world!"));
			Else If(Event Player.drawtype[2] == 1);

rule("sub - cmdpalette") {
	event { subroutine; cmdpalette; }
	actions {
		if(global.arg[0] == 0);
			event player.palette[1] += 1;
			event player.palette[1] = min(7, event player.palette[1]);
		else if(global.arg[0] == 1);
			event player.palette[1] -= 1;
			event player.palette[1] = max(0, event player.palette[1]);
		else if(global.arg[0] == 2);
			event player.palette[2] += 1;
			event player.palette[2] = min(7, event player.palette[2]);
		else if(global.arg[0] == 3);
			event player.palette[2] -= 1;
			event player.palette[2] = max(0, event player.palette[2]);
		else if(global.arg[0] == 4);
			event player.palette[3] += 1;
			event player.palette[3] = min(7, event player.palette[3]);
		else if(global.arg[0] == 5);
			event player.palette[3] -= 1;
			event player.palette[3] = max(0, event player.palette[3]);
		end;
		global.arg[0] = max(event player.palette[1],max(
					  	event player.palette[2],
						event player.palette[3]));
		if(global.arg[0] == 0);
			event player.palette[0] = custom color(0, 0, 0, 255);
		else;
			global.arg[1] = event player.palette[1] / global.arg[0] * 255;
			global.arg[2] = event player.palette[2] / global.arg[0] * 255;
			global.arg[3] = event player.palette[3] / global.arg[0] * 255;
			event player.palette[0] = custom color(
				round to integer(global.arg[1], to nearest),
				round to integer(global.arg[2], to nearest),
				round to integer(global.arg[3], to nearest),
				255);
		end;
	}
}

rule("sub - cmdemote") {
	event {Subroutine; cmdemote;}
	actions {
		If(Global.arg[0] == 0);
			If(Is Communicating Any Emote(Event Player));
				Global.tmp[0] = Filtered Array(All Players(All Teams), Is In Line of Sight(Eye Position(Current Array Element), Eye Position(
					Event Player), Barriers Do Not Block LOS) && Hero Of(Current Array Element) == Hero Of(Event Player)
					&& Is Communicating Any Emote(Current Array Element));
				For Global Variable(index, 0, Count Of(Global.tmp[0]), 1);
					If(Is Communicating(Global.tmp[0][Global.index], Emote Up));
						Cancel Primary Action(Global.tmp[0][Global.index]);
						Communicate(Global.tmp[0][Global.index], Emote Up);
					Else If(Is Communicating(Global.tmp[0][Global.index], Emote Left));
						Cancel Primary Action(Global.tmp[0][Global.index]);
						Communicate(Global.tmp[0][Global.index], Emote Left);
					Else If(Is Communicating(Global.tmp[0][Global.index], Emote Right));
						Cancel Primary Action(Global.tmp[0][Global.index]);
						Communicate(Global.tmp[0][Global.index], Emote Right);
					Else If(Is Communicating(Global.tmp[0][Global.index], Emote Down));
						Cancel Primary Action(Global.tmp[0][Global.index]);
						Communicate(Global.tmp[0][Global.index], Emote Down);
					End;
				End;
			End;
		Else If(Global.arg[0] == 1);
			If(Is Communicating Any Emote(Event Player));
				Global.tmp[0] = Filtered Array(All Players(All Teams), Is In Line of Sight(Eye Position(Current Array Element), Eye Position(
					Event Player), Barriers Do Not Block LOS) && Hero Of(Current Array Element) == Hero Of(Event Player));
				Global.tmp[0] = Mapped Array(Global.tmp[0], Array(X Component Of(Local Vector Of(Position Of(Current Array Element), Event Player,
					Rotation And Translation)), Current Array Element));
				Global.tmp[0] = Sorted Array(Global.tmp[0], Current Array Element[0]);
				Global.tmp[1] = Index Of Array Value(Mapped Array(Global.tmp[0], Current Array Element[1]), Event Player);
				Global.tmp[2] = 100;
				If(0 < Global.tmp[1]);
					Global.tmp[2] = Min(Global.tmp[2], First Of(Global.tmp[0])[0] / (0 - Global.tmp[1]));
				End;
				If(Global.tmp[1] < Count Of(Global.tmp[0]) - 1);
					Global.tmp[2] = Min(Global.tmp[2], Last Of(Global.tmp[0])[0] / (Count Of(Global.tmp[0]) - 1 - Global.tmp[1]));
				End;
				For Global Variable(index, 0, Count Of(Global.tmp[0]), 1);
					Teleport(Global.tmp[0][Global.index][1], World Vector Of((Global.index - Global.tmp[1]) * Global.tmp[2] * Left, Event Player,
						Rotation And Translation));
					Set Facing(Global.tmp[0][Global.index][1], Facing Direction Of(Event Player), To World);
					If(Is Communicating(Global.tmp[0][Global.index][1], Emote Up));
						Cancel Primary Action(Global.tmp[0][Global.index][1]);
						Communicate(Global.tmp[0][Global.index][1], Emote Up);
					Else If(Is Communicating(Global.tmp[0][Global.index][1], Emote Left));
						Cancel Primary Action(Global.tmp[0][Global.index][1]);
						Communicate(Global.tmp[0][Global.index][1], Emote Left);
					Else If(Is Communicating(Global.tmp[0][Global.index][1], Emote Right));
						Cancel Primary Action(Global.tmp[0][Global.index][1]);
						Communicate(Global.tmp[0][Global.index][1], Emote Right);
					Else If(Is Communicating(Global.tmp[0][Global.index][1], Emote Down));
						Cancel Primary Action(Global.tmp[0][Global.index][1]);
						Communicate(Global.tmp[0][Global.index][1], Emote Down);
					End;
				End;
				Global.tmp = Empty Array;
			End;
		"repeat emote"
		Else If(Global.arg[0] == 100);
			If(Event Player.subflags[3]);
				Event Player.subflags[3] = False;
				Small Message(Event Player, Custom String("hello world!"));
			Else;
				Start Rule(emoterepeat, Do Nothing);
				Small Message(Event Player, Custom String("hello world!"));
			End;
		End;
	}
}

			Event Player.oriprint[10] - Event Player.oriprint[9]) * Normalize(Vector(0 - X Component Of(Event Player.oriprint[4])
			* Y Component Of(Event Player.oriprint[4]), X Component Of(Event Player.oriprint[4]) * X Component Of(Event Player.oriprint[4])
			+ Z Component Of(Event Player.oriprint[4]) * Z Component Of(Event Player.oriprint[4]), 0 - Z Component Of(
			Event Player.oriprint[4]) * Y Component Of(Event Player.oriprint[4]))) * Event Player.oriprint[7] + Down * 1.780, True);
		Set Max Ammo(Event Player, 0, 1000);
		Set Ammo(Event Player, 0, 1000);
		Start Holding Button(Event Player, Button(Primary Fire));
		While(Event Player.oriprint[8] < Count Of(Event Player.oriprint[3]));
			Event Player.oriprint[9] = Total Time Elapsed;
			Event Player.oriprint[10] = Total Time Elapsed + Square Root((
				Event Player.oriprint[3][Event Player.oriprint[8] + 2] - Event Player.oriprint[3][Event Player.oriprint[8]]) * (
				Event Player.oriprint[3][Event Player.oriprint[8] + 2] - Event Player.oriprint[3][Event Player.oriprint[8]]) + (
				Event Player.oriprint[3][Event Player.oriprint[8] + 3] - Event Player.oriprint[3][Event Player.oriprint[8] + 1]) * (
				Event Player.oriprint[3][Event Player.oriprint[8] + 3] - Event Player.oriprint[3][Event Player.oriprint[8] + 1]))
				/ Event Player.oriprint[3][Event Player.oriprint[8] + 4];
			Wait Until(Event Player.menuopen, Event Player.oriprint[10] - Event Player.oriprint[9]);
			Log To Inspector(Event Player.oriprint[9]);
			Log To Inspector(Event Player.oriprint[10]);
			If(Event Player.menuopen);
				Break;
			End;
			Event Player.oriprint[8] += 5;
		End;
		Stop Holding Button(Event Player, Button(Primary Fire));
		Set Ammo(Event Player, 0, 150);
		Set Max Ammo(Event Player, 0, 150);
		Stop Forcing Player Position(Event Player);
		Teleport(Event Player, Event Player.oriprint[11]);
		Enable Movement Collision With Environment(Event Player);
		Stop Forcing Player To Be Hero(Event Player);
		Set Invisible(Event Player, None);
		Event Player.stat[2] = 1;
		Event Player.oriprint[0] = True;
	}
